%%%%%%%%% Learning force fields %%%%%%%%%

\nomenclature[F0]{$\rho$}{Local atomic configuration}

\nomenclature[F1]{$M$}{Number of atoms within the cutoff radius of an atomic configuration $\rho$}

\nomenclature[F2]{$\varepsilon^r \mid \varepsilon(\rho)$}{Reference and latent local energy function}

\nomenclature[F21]{$ \hat{\varepsilon}(\rho) \mid \hat{\sigma}^2(\rho)$}{Mean and variance of the local energy predictive distribution}

\nomenclature[F22]{$k(\rho, \rho')$}{Local energy kernel evaluated for configurations $\rho$ and $\rho'$}

\nomenclature[F22]{$k_n \mid k_{n}^s \mid k_n^{\neg u}$}{Local energy $n$-body kernel,  symmetric $n$-body kernel and symmetric $n$-body kernel that is non unique}

\nomenclature[F22]{$k_{MB} \mid k_{MB}^s \mid k_{MB}^{ds}$}{Local energy many-body kernels,  many-body kernel symmetric over the full orthogonal group and many-body kernel symmetric over a discrete group}



\nomenclature[F3]{$\mathbf{f}^r \mid \mathbf{f}(\rho)$}{Reference and latent force on the central atom of a configuration}

\nomenclature[F31]{$\hat{\mathbf{f}}(\rho) \mid \hat{\mathbf{\Sigma}}(\rho)$}{Mean and covariance of the force predictive distribution}

\nomenclature[F32]{$\mathbf{K}(\rho, \rho')$}{Matrix-valued force kernel evaluated for two configurations}

\nomenclature[F32]{$\mathbf{K}_n^{G} \mid \mathbf{K}_{MB}^{G}$}{Matrix-valued $n$-body and many-body force kernels that are covariant over the group $G$}

% groups
\nomenclature[F40]{$\mathcal{P} \mid \mathbf{P}$}{Abstract operator and matrix representation of a permutation of atoms of the same chemical species}
\nomenclature[F41]{$\mathcal{R} \mid \mathbf{R}$}{Abstract operator and matrix representation of a rotation}
\nomenclature[F42]{$\mathcal{F} \mid \mathbf{F}$}{Abstract operator and matrix representation of a reflection}
\nomenclature[F43]{$\mathcal{Q} \mid \mathbf{Q}$}{Abstract operator and matrix representation of an element of the orthogonal group (either a rotation or a reflection)}
\nomenclature[F44]{$C_n \mid D_n$}{Cyclic and dihedral group of order $n$}
